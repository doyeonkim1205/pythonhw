#include <stdio.h>

int main()   

{
   int a, b, c;
   int result = 0;
   
    printf("숫자를 입력하시오(3개 입력)\n");
    scanf("%d %d %d", &a, &b, &c);
    
    if (c == 0) {
        printf("0으로 나눌 수 없습니다. 다시 입력하시오.");
        return 1;
    } 
    
    result = (a + b) * c - (a - b) / c;
    printf("(%d + %d) X %d - (%d - %d) / %d = %d ",a, b, c, a, b, c, result);
    return 0;
}
