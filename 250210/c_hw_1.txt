#include <stdio.h>

void main()   
{
   int a, b, c;
   float result = 0;
   
    printf("숫자를 입력하시오(3개 입력)\n");
    scanf("%d %d %d", &a, &b, &c);
    
    if (c == 0) {
        printf("0으로 나눌 수 없습니다. 다시 입력하시오.");
    } 
    result = (a + b) * c - ((a - b) / c);
    printf("(%d + %d) X %d - (%d - %d) / %d = %.1f ",a, b, c, a, b, c, result);
}
