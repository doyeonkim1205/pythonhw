#include <stdio.h>

int main()   
{
   int i = 0;
   int result = 0;
   
    printf("세자리 숫자를 입력하시오\n");
    scanf("%d", &i);
    
    int Iii = i / 100;
    int iIi = (i / 10) % 10;
    int iiI = i % 10;
    result = Iii + iIi + iiI;
    
    printf("자리 수의 합 : %d", result);
    return 0;
}
